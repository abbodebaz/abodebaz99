```tsx
'use client'
import { useState } from 'react'
import dynamic from 'next/dynamic'
import Intro from '@/components/Intro'
import Navbar from '@/components/layout/Navbar'
import Hero from '@/components/sections/Hero'
import Footer from '@/components/layout/Footer'

const About = dynamic(() => import('@/components/sections/About'))       // ← جديد
const Journey = dynamic(() => import('@/components/sections/Journey'))
const Systems = dynamic(() => import('@/components/sections/Systems'))
const Achievements = dynamic(() => import('@/components/sections/Achievements'))
const Impact = dynamic(() => import('@/components/sections/Impact'))
const Framework = dynamic(() => import('@/components/sections/Framework'))
const Contact = dynamic(() => import('@/components/sections/Contact'))

export default function Home() {
  const [introComplete, setIntroComplete] = useState(false)

  return (
    <>
      {!introComplete && <Intro onComplete={() => setIntroComplete(true)} />}
      <main style={{ opacity: introComplete ? 1 : 0, transition: 'opacity 0.8s ease' }}>
        <Navbar />
        <Hero />
        <About />          {/* ← جديد — بين Hero وJourney */}
        <Journey />
        <Systems />
        <Achievements />
        <Impact />
        <Framework />
        <Contact />
        <Footer />
      </main>
    </>
  )
}
```

## PART 6 — إعداد next.config.mjs للصور الخارجية

تأكد إن `next.config.mjs` يحتوي على:

```js
/** @type {import('next').NextConfig} */
const nextConfig = {
  images: {
    remotePatterns: [
      {
        protocol: 'https',
        hostname: 'images.unsplash.com',
      },
    ],
  },
}

export default nextConfig
```

هذا يسمح بتحميل صور Unsplash كـ fallback + صور `/public` المحلية.

## تعليمات رفع الصور — بعد تطبيق الكود

### الصورة الشخصية:
1. أضف صورتك الاحترافية في `/public/images/profile.jpg`
2. الأبعاد المثالية: **400×500px** أو أكبر
3. Format: JPG أو WebP

### Screenshots الأنظمة:
أضف الصور بهذه الأسماء في `/public/images/`:

| اسم الملف | وش يكون فيه |
|---|---|
| `system-ecommerce.jpg` | screenshot المتجر أو dashboard المبيعات |
| `system-delivery.jpg` | screenshot نظام التوصيل |
| `system-crm.jpg` | screenshot الـ CRM أو pipeline |
| `system-ai.jpg` | screenshot workflow الـ AI |

**الأبعاد المثالية للـ Screenshots:** 1280×800px أو 16:9

> 💡 لو ما عندك screenshot معين، خلّ المسار كما هو — الكود يرجع تلقائياً لصورة Unsplash كـ fallback.
