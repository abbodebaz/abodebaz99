import { useState, useEffect, useCallback } from 'react'

// ======================================================
// التوقيت الكامل — بطيء ومريح للعين
// ======================================================
// 0.0s  شاشة سوداء فارغة
// 1.0s  يظهر اليوم والوقت
// 2.5s  إشعار 1
// 4.2s  إشعار 2
// 5.9s  إشعار 3
// 7.6s  إشعار 4
// 9.5s  تخفت الإشعارات + "هذا الصباح يتكرر"
// 12.0s اسم عبدالرحمن
// 15.0s fade out
// 16.0s onComplete

export default function Intro({ onComplete }: { onComplete: () => void }) {
  const [phase, setPhase] = useState(0)
  const [visibleCount, setVisibleCount] = useState(0)
  const [dayData, setDayData] = useState(DAY_DATA[1])
  const stableOnComplete = useCallback(onComplete, [onComplete])

  useEffect(() => {
    const today = new Date().getDay()
    setDayData(DAY_DATA[today])
  }, [])

  useEffect(() => {
    const t: ReturnType<typeof setTimeout>[] = []

    t.push(setTimeout(() => setPhase(1), 1000))          // يظهر اليوم
    t.push(setTimeout(() => setVisibleCount(1), 2500))   // إشعار 1
    t.push(setTimeout(() => setVisibleCount(2), 4200))   // إشعار 2
    t.push(setTimeout(() => setVisibleCount(3), 5900))   // إشعار 3
    t.push(setTimeout(() => setVisibleCount(4), 7600))   // إشعار 4
    t.push(setTimeout(() => setPhase(2), 9500))          // "يتكرر كل أسبوع"
    t.push(setTimeout(() => setPhase(3), 12000))         // اسم عبدالرحمن
    t.push(setTimeout(() => setPhase(4), 15000))         // fade out
    t.push(setTimeout(() => stableOnComplete(), 16000))  // انتهى

    return () => t.forEach(clearTimeout)
  }, [stableOnComplete])

  return (
    <AnimatePresence>
      {phase < 4 && (
        <motion.div
          exit={{ opacity: 0 }}
          transition={{ duration: 1.0, ease: 'easeInOut' }}
          className="fixed inset-0 z-[9999] flex flex-col items-center justify-center overflow-hidden"
          style={{ background: '#050810' }}
        >
          {/* خلفية — نقاط ضوئية */}
          <div
            className="absolute inset-0 pointer-events-none"
            style={{
              backgroundImage: 'radial-gradient(rgba(59,130,246,0.07) 1px, transparent 1px)',
              backgroundSize: '52px 52px',
            }}
          />
          {/* glow مركزي */}
          <div
            className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[500px] h-[500px] rounded-full pointer-events-none"
            style={{
              background: 'radial-gradient(circle, rgba(59,130,246,0.07) 0%, transparent 70%)',
              filter: 'blur(60px)',
            }}
          />

          {/* ===== المرحلة 1+2: الإشعارات ===== */}
          <AnimatePresence>
            {phase >= 1 && phase < 3 && (
              <motion.div
                key="notifications"
                initial={{ opacity: 0 }}
                animate={{ opacity: phase === 2 ? 0.15 : 1 }}
                transition={{ duration: 1.2, ease: 'easeInOut' }}
                className="flex flex-col items-center gap-4 w-full px-6 max-w-md"
              >
                {/* اليوم والوقت */}
                <motion.div
                  initial={{ opacity: 0, y: -8 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ duration: 0.8, ease: 'easeOut' }}
                  className="flex items-center gap-3 mb-2"
                >
                  <motion.div
                    animate={{ opacity: [1, 0.3, 1] }}
                    transition={{ duration: 2, repeat: Infinity }}
                    className="w-2 h-2 rounded-full"
                    style={{ background: '#3B82F6', boxShadow: '0 0 10px rgba(59,130,246,0.9)' }}
                  />
                  <span
                    className="text-xs uppercase tracking-[0.35em]"
                    style={{ color: 'rgba(59,130,246,0.85)', fontFamily: 'Sora, sans-serif' }}
                  >
                    {dayData.day} — {dayData.time}
                  </span>
                </motion.div>

                {/* الإشعارات — تظهر واحدة بعد واحدة */}
                <div className="flex flex-col gap-3 w-full">
                  {dayData.messages.map((msg, i) => (
                    <AnimatePresence key={i}>
                      {visibleCount > i && (
                        <motion.div
                          initial={{ opacity: 0, y: 20, scale: 0.97 }}
                          animate={{ opacity: 1, y: 0, scale: 1 }}
                          transition={{ duration: 0.7, ease: [0.22, 1, 0.36, 1] }}
                          className="flex items-start gap-3 px-4 py-3.5 rounded-2xl"
                          style={{
                            background: 'rgba(255,255,255,0.055)',
                            border: '1px solid rgba(255,255,255,0.09)',
                            backdropFilter: 'blur(12px)',
                            boxShadow: '0 4px 24px rgba(0,0,0,0.3)',
                          }}
                        >
                          <span className="text-base mt-0.5 shrink-0 select-none">{msg.icon}</span>
                          <p
                            className="text-sm leading-relaxed"
                            style={{
                              color: 'rgba(255,255,255,0.8)',
                              fontFamily: 'DM Sans, sans-serif',
                              direction: 'rtl',
                            }}
                          >
                            {msg.text}
                          </p>
                        </motion.div>
                      )}
                    </AnimatePresence>
                  ))}
                </div>
              </motion.div>
            )}
          </AnimatePresence>

          {/* ===== المرحلة 2: "يتكرر كل أسبوع" فوق الإشعارات الخافتة ===== */}
          <AnimatePresence>
            {phase === 2 && (
              <motion.div
                key="repeat-text"
                initial={{ opacity: 0, y: 16 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0 }}
                transition={{ duration: 1.0, ease: [0.22, 1, 0.36, 1] }}
                className="absolute inset-0 flex flex-col items-center justify-center px-8 text-center pointer-events-none"
              >
                <motion.p
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  transition={{ delay: 0.3, duration: 1.0 }}
                  className="text-xl md:text-2xl mb-3"
                  style={{
                    color: 'rgba(255,255,255,0.6)',
                    fontFamily: 'DM Sans, sans-serif',
                    fontWeight: 400,
                    lineHeight: 1.7,
                    direction: 'rtl',
                  }}
                >
                  هذا الصباح يتكرر كل أسبوع.
                </motion.p>
                <motion.p
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  transition={{ delay: 1.0, duration: 1.0 }}
                  className="text-xl md:text-2xl"
                  style={{
                    color: 'rgba(255,255,255,0.4)',
                    fontFamily: 'DM Sans, sans-serif',
                    fontWeight: 400,
                    direction: 'rtl',
                  }}
                >
                  في معظم الشركات.
                </motion.p>
              </motion.div>
            )}
          </AnimatePresence>

          {/* ===== المرحلة 3: اسم عبدالرحمن ===== */}
          <AnimatePresence>
            {phase === 3 && (
              <motion.div
                key="name-reveal"
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                exit={{ opacity: 0 }}
                transition={{ duration: 1.0 }}
                className="absolute inset-0 flex flex-col items-center justify-center px-8 text-center"
              >
                {/* خط أزرق يمتد */}
                <motion.div
                  initial={{ scaleX: 0, opacity: 0 }}
                  animate={{ scaleX: 1, opacity: 1 }}
                  transition={{ duration: 0.8, ease: [0.22, 1, 0.36, 1] }}
                  className="mb-10 origin-left"
                  style={{
                    width: '48px',
                    height: '1px',
                    background: '#3B82F6',
                    boxShadow: '0 0 16px rgba(59,130,246,0.9)',
                  }}
                />

                {/* الاسم */}
                <motion.h1
                  initial={{ opacity: 0, y: 24, filter: 'blur(10px)' }}
                  animate={{ opacity: 1, y: 0, filter: 'blur(0px)' }}
                  transition={{ delay: 0.4, duration: 1.0, ease: [0.22, 1, 0.36, 1] }}
                  style={{
                    fontFamily: 'Sora, sans-serif',
                    fontSize: 'clamp(28px, 5.5vw, 68px)',
                    fontWeight: 700,
                    color: 'white',
                    letterSpacing: '-0.02em',
                    textShadow: '0 0 80px rgba(59,130,246,0.35)',
                    lineHeight: 1.1,
                    marginBottom: '20px',
                  }}
                >
                  Abdulrahman Bazarah
                </motion.h1>

                {/* الرسالة */}
                <motion.p
                  initial={{ opacity: 0, y: 14 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ delay: 1.0, duration: 0.9 }}
                  style={{
                    fontFamily: 'DM Sans, sans-serif',
                    fontSize: 'clamp(14px, 2vw, 19px)',
                    color: 'rgba(255,255,255,0.42)',
                    letterSpacing: '0.01em',
                    direction: 'rtl',
                  }}
                >
                  أبني الأنظمة التي تنهي هذا الصباح.
                </motion.p>

                {/* خط سفلي ناعم */}
                <motion.div
                  initial={{ opacity: 0, scaleX: 0 }}
                  animate={{ opacity: 1, scaleX: 1 }}
                  transition={{ delay: 1.6, duration: 0.8 }}
                  className="mt-8"
                  style={{
                    width: '80px',
                    height: '1px',
                    background: 'linear-gradient(to right, transparent, rgba(59,130,246,0.5), transparent)',
                  }}
                />
              </motion.div>
            )}
          </AnimatePresence>

        </motion.div>
      )}
    </AnimatePresence>
  )
}
```

## التوقيت الجديد — بالثواني

```
0.0s   شاشة سوداء فارغة
1.0s   يظهر اليوم والوقت (مثال: Monday — 8:47 AM)
2.5s   إشعار 1 يظهر ببطء
4.2s   إشعار 2 يظهر
5.9s   إشعار 3 يظهر
7.6s   إشعار 4 يظهر (الزائر يقرأ كلهم)
9.5s   الإشعارات تخفت + "هذا الصباح يتكرر كل أسبوع"
12.0s  Abdulrahman Bazarah + الرسالة
15.0s  fade out ناعم
16.0s  الصفحة الرئيسية
```
